?Create a new project
    docker build -t rails-toolbox-itbookstorejenkins --build-arg USER_ID=$(id -u) --build-arg GROUP_ID=$(id -g) -f Dockerfile.rails .
    docker run -it --rm -v $PWD:/opt/app rails-toolbox-itbookstorejenkins rails new -d postgresql --skip-bundle itbookstorejenkins

?Create the development environment
    Move docker-entrypoint.sh and wait-for inside the itbookstorejenkins folder
        chmod +x itbookstorejenkins/docker-entrypoint.sh #make it executable
    *Postgresql
        in database.yml modificari:
            in default:
                ...
                pool: <%= ENV.fetch("RAILS_MAX_THREADS") { 5 } %>
                host: <%= ENV.fetch("DATABASE_HOST"){ 'none'}  %>
                username: <%= ENV.fetch("POSTGRES_USER"){ 'none'}  %>
                password: <%= ENV.fetch("POSTGRES_PASSWORD"){ 'none'}  %>
                database: <%= ENV.fetch("POSTGRES_DB"){ 'none'}  %>
                timeout: 5000
            in development sa ramana doar <<: *default
            in production sa ramana doar <<: *default
    cd ..
    cd itbookstorejenkins
    docker-compose run --rm --user "$(id -u):$(id -g)" web_itbookstorejenkins bin/rails webpacker:install
    in config/webpacker.yml
        development:
            ...
            dev_server:
                ...
                port: 3036 #change this
                public: localhost:3036 #change this
                ...
                watch_options:
                    ...
                    poll: true
    in config/environments/development
	    comment config.action_view.annotate_rendered_view_with_filenames = true
    cd ..
    cd itbookstorejenkins
    change in docker-compose.yml 
        services:
            web_aplicatie2:
                ports:
                - "3001:3000" #change only external port
            webpack_dev_server_aplicatie2:
                ports: 
                - "3036:3036" #change this
    in config/environments/development add
	config.hosts << "itbookstorejenkins.local"
    in Windows/system32/drivers/etc/hosts
	add 127.65.43.next itbookstorejenkins.local
    in admin elevated cmd 
	netsh interface portproxy add v4tov4 listenport=80 listenaddress=127.65.43.next connectport=300next connectaddress=127.0.0.1
    docker-compose up -d

?Use it in development
    Generate a scaffold
        docker-compose exec --user "$(id -u):$(id -g)" web_itbookstorejenkins bin/rails g scaffold User first_name last_name
        docker-compose exec --user "$(id -u):$(id -g)" web_itbookstorejenkins bin/rails db:migrate
    Adding a gem
        docker-compose exec web_itbookstorejenkins bundle install

?add redis
    in docker-compose.yml
        redis_itbookstorejenkins:
            image: redis
    in docker-compose-itbookstorejenkins.yml
        redis_itbookstorejenkins:
            image: redis
            restart: always
    in Gemfile uncomment redis and run bundle install
    in cable.yml or wherever is the path for redis server change to redis_itbookstorejenkins:6379
    and also in production you need to connect the redis container to network
